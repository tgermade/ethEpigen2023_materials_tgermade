suppressPackageStartupMessages({
  library(epiwraps)
  library(AnnotationHub)
  library(MotifDb)
  library(memes)
  library(universalmotif)
  library(ensembldb)
  library(ggplot2)
  library(ATACseqQC)
})

ah <- AnnotationHub(localHub=TRUE)
ensdb <- ah[["AH89211"]] # mouse ensembldb object

Prepare the data

download.file("https://ethz-ins.org/content/w7/atac.chr19.bam", 
              "atac.chr19.bam", mode="wb")
download.file("https://ethz-ins.org/content/w7/atac.chr19.bam.bai", 
              "atac.chr19.bam.bai", mode="wb")
bam <- "atac.chr19.bam"


# create a track using all fragments, computing the coverage at each position (and force Ensembl naming)
bam2bw(bam, output_bw = "full_cov.bw", paired=TRUE, binWidth=10L, forceSeqlevelsStyle = "Ensembl")

# create a track using all fragments, but computing the number of cuts/insertion sites at each position
epiwraps::bam2bw(bam, output_bw = "full_cuts.bw", paired=TRUE, binWidth=1L, type="ends", 
                 shift=c(4L,-5L), forceSeqlevelsStyle = "Ensembl")

# create a track using only nucleosome-free fragments, computing coverage
bam2bw(bam, output_bw = "NF_cov.bw", paired=TRUE, binWidth=5L, minFragLength=30, 
       maxFragLength=120, forceSeqlevelsStyle = "Ensembl")

# create a track using only nucleosome-free fragments, the number of cuts/insertion sites at each position
bam2bw(bam, output_bw = "NF_cuts.bw", paired=TRUE, binWidth=1L, type="ends", minFragLength=30, 
       maxFragLength=120, shift=c(4L,-5L), forceSeqlevelsStyle = "Ensembl")

# create a track using only mono-nucleosome fragments, computing coverage
bam2bw(bam, output_bw = "mono_cov.bw", paired=TRUE, binWidth=10L, minFragLength=140,
       maxFragLength=220, forceSeqlevelsStyle = "Ensembl")

# create a track using only mono-nucleosome fragments, computing centers
bam2bw(bam, output_bw = "mono_centers.bw", paired=TRUE, binWidth=5L, minFragLength=140,
       maxFragLength=220, type="center", forceSeqlevelsStyle = "Ensembl")
# we prepare the list of tracks
tracks <- c("Full coverage"="full_cov.bw", "NF coverage"="NF_cov.bw", "All cuts"="full_cuts.bw", "NF cuts"="NF_cuts.bw", "Mono cov"="mono_cov.bw", "Mono centers"="mono_centers.bw")

Import genome

genome <- ah[["AH68356"]]
# get the sequence for chr19:
chr19 <- rtracklayer::import(genome)["19"]

Footprint factor 1: JUND

# get JUND motif
q.jun <- query(MotifDb, c("JUND","Mus"))
motif.jun <- q.jun[["Mmusculus-HOCOMOCOv10-JUND_MOUSE.H10MO.A"]]
motif.jun2 <- convert_motifs(motif.jun, class="TFBSTools-PFMatrix")

# find motif occurences
moi.jun <- motifmatchr::matchMotifs(motif.jun2, chr19, out="positions", p.cutoff=1e-5)[[1]]
# convert to GRanges (when using motifmatchr)
moi.jun <- as(setNames(moi.jun, names(chr19)), "GRanges")
# extract signals around the motif occurences
# we zoom in to 500bp around the motif centers, in windows of 5bp
sm.jun <- signal2Matrix(tracks, moi.jun, w=5, extend=500)
nf.jun <- bwNormFactors(tracks, useSeqLevels="19", nwind=5000L)
# then we apply the normalization factors:
smb.jun <- rescaleSignalMatrices(sm.jun, nf.jun)
plotEnrichedHeatmaps(smb.jun, trim=0.95, minRowVal = 15, colors = c("white","darkred"))

epiwraps::motifFootprint("atac.chr19.bam", motif.jun, moi.jun)

Footprint factor 2: STAT5

# get STAT5 motif
q.stat <- query(MotifDb, c("stat5","Mus"))
motif.stat <- q.stat[[1]]
motif.stat2 <- convert_motifs(motif.stat, class="TFBSTools-PFMatrix")

# find motif occurences
moi.stat <- motifmatchr::matchMotifs(motif.stat2, chr19, out="positions", p.cutoff=1e-5)[[1]]
# convert to GRanges (when using motifmatchr)
moi.stat <- as(setNames(moi.stat, names(chr19)), "GRanges")
# extract signals around the motif occurences
# we zoom in to 250bp around the motif centers, in windows of 5bp
sm.stat <- signal2Matrix(tracks, moi.stat, w=5, extend=250)
nf.stat <- bwNormFactors(tracks, useSeqLevels="19", nwind=5000L)
# then we apply the normalization factors:
smb.stat <- rescaleSignalMatrices(sm.stat, nf.stat)
plotEnrichedHeatmaps(smb.stat, trim=0.95, minRowVal = 0, colors = c("white","darkred"))

epiwraps::motifFootprint("atac.chr19.bam", motif.stat, moi.stat)

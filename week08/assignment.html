Get the data

download.file("https://ethz-ins.org/content/mouse_mm38_hippocampus.peakCounts.SE.rds", "mouse_mm38_hippocampus.peakCounts.SE.rds", mode="wb")
se <- readRDS("mouse_mm38_hippocampus.peakCounts.SE.rds")
motifs <- query(MotifDb, c("HOCOMOCOv10", "Mmusculus"))
motifs <- do.call(TFBSTools::PWMatrixList, setNames(
           universalmotif::convert_motifs(motifs, class="TFBSTools-PWMatrix"),
           mcols(motifs)$providerName))
dir.genome <- "/reference/Mus_musculus/Ensembl/GRCm38/Sequence/WholeGenomeFasta/Mus_musculus.GRCm38.dna_sm.primary_assembly.fa"
genome <- open(Rsamtools::FaFile(dir.genome))
se$condition <- c(
  rep("CTRL", sum(grepl("Control", rownames(colData(se))))), 
  rep("FSS", sum(grepl("FSS", rownames(colData(se)))))
)
se$sexMale <- c(
  grepl("_male", rownames(colData(se)))
)
se <- chromVAR::addGCBias(se, genome=genome)
# we find which peaks contain which motifs
# we don't use `out="positions"` 
moi <- motifmatchr::matchMotifs(motifs, subject=se, genome=genome)


dev <- chromVAR::computeDeviations(object = se, annotations=moi,
                                   background_peaks=getBackgroundPeaks(se, niterations=1000))

# variability <- computeVariability(dev)
# plotVariability(variability)
assays(dev)$norm <- scale(assays(dev)$z)

We wanted to check the what TF motifs are enriched in the peaks we get from ATACseq analysis of mice that underwent a forced swim test. Mice that underwent the treatment show putatively increased DNA accessibility in some stress-related steroid receptors among others when compared to untreated mice. Additionally, we controlled for sex and found a set of TFs that show a putative accessibility decrease in male mice compared to female.
